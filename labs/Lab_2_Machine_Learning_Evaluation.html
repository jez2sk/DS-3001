# calculates  precision, sensitivity, specificity, and F1 scores
def metrics_finder(y_true, y_pred):
    precision = precision_score(y_true, y_pred)
    sensitivity = recall_score(y_true, y_pred)
    specificity = recall_score(np.logical_not(y_true), np.logical_not(y_pred))
    f1 = f1_score(y_true, y_pred)
    return precision, sensitivity, specificity, f1

# add to table
