# Extract all list text
list_items = soup.find_all('li')
text = " ".join([li.get_text() for li in list_items])

#choose which solar system objects to scan for
search = ["Mercury", "Venus", "Earth", "Mars", "Jupiter", "Saturn", "Uranus", "Neptune"]

# Create a regex pattern dynamically from the list of object names
search_pattern = "|".join(search)  # Join planet names with "|"
pattern = rf"({search_pattern})\s*\(([\d\.]+)\s*–\s*([\d\.]+)\s*AU"

# Find all matches
matches = re.findall(pattern, text)

# Store extracted data
data = [{"Planet": planet, "Min Distance (AU)": float(min_dist), "Max Distance (AU)": float(max_dist)} for planet, min_dist, max_dist in matches]

# Create a DataFrame
df = pd.DataFrame(data)

# Print the DataFrame
print("\nExtracted DataFrame:")
print(df)

From the distrbution of data, we can see that planets closer to the Sun are less spread out, while those past the asteroid belt, the outer planets, are significantly more spread out. Beyond that, we can also see that distance from the Sun may have an effect on the ecentricity of a planet's orbit—the inner planets have less variation in their orbits than the outer ones.
